## Install

NB: ShardDocs relies on [react router](https://reacttraining.com/react-router/web/guides/quick-start) for routing.

```
npm install @fa-repo/shard-docs react-router-dom
```

## Basic usage

```jsx
// index.js
import React from "react";
import { render } from "react-dom";

import ShardDocs from "@fa-repo/shard-docs";
import "@fa-repo/shard-docs/dist/index.css";

const content = [
  { type: "document", title: "Install", document: <><h1>Install</h1></> },
  { type: "document", title: "Basic usage", document: <><h1>Basic usage</h1></> },
  {
    type: "folder",
    title: "Examples",
    items: [
      { type: "document", title: "Use case A", document: <><h1>Use case A</h1></> },
      { type: "document", title: "Use case B", document: <><h1>Use case B</h1></> }
    ]
  },
  { type: "external-link", title: "Github", externalLink: "https://github.com" },
]

render(
  <ShardDocs
    title="ShardDocs demo"
    description="A description describing the purpose of the docs."
    content={content}
  />,
  document.getElementById("root")
);
```
